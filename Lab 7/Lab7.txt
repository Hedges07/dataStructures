void insertionSort(int a[], int listSize) {
    int j,temp; 
    for(int i = 1; i <listSize; i++) { 
        temp = a[i]; //save value of a[i]
        j = i - 1;
        while (j >= 0 && a[j] > temp) {  //search for location of a[i]
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = temp;
    }
}

void quickSort(int a[], int p, int q) {
    if(p < q) {
        int r = partition(a,p,q); //calls the partition()
        quickSort(a,p,r); //recursively calls quicksort
        quickSort(a,r+1,q);
    }
}

int partition(int a[], int p, int q){
    int i = p-1;
    int j = q+1;
    int x = a[p];
    int flag = 1;
    int temp;

    while(flag){
        // move left pointer
        do i = i+1;
        while (a[i]<x);
        //move right pointer
        do j = j-1;
        while (a[j]>x);
        if (i<j){
            //exchange a[1] and a[j]
            temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
        else flag = 0;
    }
    return j;
